Create a Kubernetes Pod named ckad15-memory, with a container named ckad15-memory running the polinux/stress image, and configure it to use the following specifications: Command: stress Arguments: ["--vm", "1", "--vm-bytes", "10M", "--vm-hang", "1"] Requested memory: 10Mi Memory limit: 10Mi

apiVersion: v1
kind: Pod
metadata:
  name: ckad15-memory
spec:
  containers:
  - name: ckad15-memory
    image: polinux/stress
    command: ["stress"]
    args: ["--vm", "1", "--vm-bytes", "10M", "--vm-hang", "1"]
    resources:
      requests:
        memory: "10Mi"
      limits:
        memory: "10Mi"
